prontera,133,188,6	script	Platinum Skill	735,{
	mes "[Platinum Skill]";
	mes "Eu posso lhe dar as habilidades especiais disponíveis para o seu trabalho.";
	mes "Gostaria dessas habilidades agora?";
	next;
	if (select("Sim por favor:Não") == 2) {
		mes "[Platinum Skill]";
		mes "Tenha um bom dia... >.>";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[Platinum Skill]";
	mes "Ai está!";
	close;
}
