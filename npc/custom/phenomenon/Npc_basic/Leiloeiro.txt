	mes "[Leiloeiro]";
	set @num,0;
// These few line below are to recover item that would be lost due to a new auction starting
L_RNameLoop:
	if(strcharinfo(0)==$ANamesB$[@num] && $paidB[@num]==0) goto L_RBackup;
	set @num,@num+1;
	if(@num<10) goto L_RNameLoop;
	if($Auction==1) goto L_Register;
	if($Auction==2) goto L_Bid;
	if($Auction==3) goto L_PayPickup;

// After a server restart, and no auction value is set, it will run this
// People who placed items in there before the restart should still be to retrieve there items

	mes "Você está habilitado para vender e comprar coisas dentro do leilão, infelizmente nada foi iniciado, você precisa esperar para o leilão começar.";

	mes "Eu sinto muito se você vier a perder itens";
	close;

	mes "Bem-vindo ao leilão, você pode registrar seus itens agora.";
	next;
	set @num,0;

	if(strcharinfo(0)==$ANames$[@num]) goto L_Already;

	if(getarraysize($AItems)==10) goto L_AFull;

L_End:
	mes "[Leiloeiro]";
	mes "Ok, volte quando quiser";
	close;

	mes "[Leiloeiro]";

	next;
	set @array,0;
	set @item,500;
//	set @loopcount,0;
L_ILoop:
//	set @loopcount,@loopcount+1;
//	debugmes @loopcount;
	set @item,@item+1;
	if(@item>=13005) goto L_Menu;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	set @item,@item+1;
	if(countitem(@item)>0) set @item2[@array],@item;
	if(countitem(@item)>0) set @array,@array+1;
	if(@array==10) goto L_Menu;
	goto L_ILoop;
L_Menu:
	menu getitemname(@item2[0]),-,getitemname(@item2[1]),-,getitemname(@item2[2]),-,getitemname(@item2[3]),-,getitemname(@item2[4]),-,getitemname(@item2[5]),-,getitemname(@item2[6]),-,getitemname(@item2[7]),-,getitemname(@item2[8]),-,getitemname(@item2[9]),-,"Próxima Página",L_WipeLoop;
	set @setitem,@item2[@menu-1];
	set @num,0;
L_SorryLoop:
	set @num,@num+1;
	if(@menu==@num && @setitem == 0) goto L_Sorry;
	if(@num==10) goto L_FinalReg;
	goto L_SorryLoop;
L_FinalReg:
	mes "[Leiloeiro]";
	mes "Por favor lembre que se estes itens contém cartas, ou é quebrado, estas características vão ser perdidas quando for posto pra leilão.";

	mes " ";
	mes "Então você quer registrar seu "+getitemname(@setitem);
	next;
	menu "Sim, por favor",-,"Não",L_End;

	set @amount,1;
	if(countitem(@setitem)==1) goto L_FinalInput;

	mes "Espere um minuto, parece que você tem mais de uma unidade deste item, por favor coloque a quantidade que você quer pôr no leilão.";
	next;
	input @amount;
	if(countitem(@setitem)<@amount) goto L_NotEnough;
L_FinalInput:
	set $AItems[$array],@setitem;
	set $ANames$[$array],strcharinfo(0);
	set $AAmount[$array],@amount;
	set $array,$array+1;
	delitem @setitem,@amount;

L_AFull:
	mes "[Leiloeiro]";
	mes "Oh, desculpe, parece que meu leilão está lotado neste primeiro ciclo, por favor tente na próxima.";
	close;
L_WipeLoop:
	deletearray @item2[0],10;
	set @array,0;
	goto L_ILoop;

	mes "Parece que é tudo que você tem a me oferecer, volte quando você souber o que você quer vender.";

	mes "Parece que você já tem um item registrado aqui, você gostaria de tê-lo de volta?";

L_AItemsLoop:
	set $AItems[@num],$AItems[@num+1];
	set $ANames$[@num],$ANames$[@num+1];
	set $AAmount[@num],$AAmount[@num+1];
	set @num,@num+1;
	if(@num<10) goto L_AItemsLoop;
	set $array,$array-1;
	set @num,0;
	mes "[Leiloeiro]";
	mes "Ok, todos os itens presentes e contados, tenha um bom dia.";
	close;
L_NotEnough:
	mes "[Leiloeiro]";
	mes "Parece que você não tem muitos "+getitemname(@setitem)+"'s";
	close;
// If the Auction has enterd phase 2, it will run this, during this time items you can bid on registered
// No items can be retrieved during this time, they are now stuck

L_Bid:
	mes "Em que você gostaria de dar um lance?";
	next;
	set @num,0;

	set @menu$[@num],$AAmount[@num]+" x "+getitemname($AItems[@num])+" "+$bid[@num]+"z";

	@menu$[5],-,

	@menu$[7],-,
	@menu$[8],-,

	mes "O lance atual está em "+$bid[@num]+"z";
	mes "Foi feito por "+$bidders$[@num];
	if($bidders$[@num]==strcharinfo(0)) goto L_AlreadyBid;
	mes "Quando você quer ofertar?";
	mes "(Use 0 para cancelar a operação)";
	next;
	input @bid;
	if(@bid<=0) goto L_End;
	if(zeny<@bid) goto L_LowZeny;
	if(@bid<$bid[@num]) goto L_LowBid;
	mes "[Leiloeiro]";
	mes "Você está para ofertar "+@bid+"z para "+$AAmount[@num]+" x "+getitemname($AItems[@num]);
	mes "Você está certo disso?";
	next;
	menu "Sim",-,"Não",L_End;
	set $bid[@num],@bid;
	set $bidders$[@num],strcharinfo(0);
	mes "[Leiloeiro]";
	mes "Seu lance foi dado";
	close;
L_LowZeny:
	mes "[Leiloeiro]";
	mes "Desculpe, você precisa me mostrar que tem recursos para poder pagar o lance caso você ganhe.";
	mes "Por favor pegue mais dinheiro, ou dê um lance menor.";
	close;
L_CantBid:
	mes "Desculpe, você não pode dar um lance em seus próprios itens";
	close;
L_LowBid:
	mes "[Leiloeiro]";
	mes "Seu lance parece ser menor do que o atual, tente dar um lance com um número maior de zeny";
	next;
	mes "[Leiloeiro]";
	goto L_ReBid;
L_AlreadyBid:
	mes "Ei espere, este é você, você não pode dar um lance novamente.";
	close;
// If the Auction has entered phase 3, it will run this
// During this time, items that have been bid on can be bought
// Items not bid on can be picked back up by the seller
// And if any of the items have been bought, and are paid for, seller can pick up their money

L_PayPickup:
	mes "O leilão agora acabou.";
	mes "O que você gostaria de fazer?";
	next;
	menu "Checar os itens que vendi",-,"Checar os itens que ganhei",L_Won,"Nada",L_End;

// Check starts here for if you have sold anything during this auction

	set @num,0;
L_SoldLoop:
	if($ANames$[@num]==strcharinfo(0)) goto L_Sold;
	set @num,@num+1;
	if(@num<10) goto L_SoldLoop;
	mes "[Leiloeiro]";
	mes "Desculpe, parece que você pegou todos os seus itens, dinheiro";
	mes "ou você não registrou nenhum item neste leilão.";
	close;
L_Sold:
	mes "[Leiloeiro]";
	if($paid[@num]==3) goto L_Returned;
	mes "Ah, eu vejo aqui, você vendeu "+$AAmount[@num]+" x "+getitemname($AItems[@num]);
	mes "Deixe-me ver para quem foi vendido...";
	next;
	if($bid[@num]==0) goto L_GiveBack;
	if($paid[@num]==2) goto L_AlreadyPaid;
	mes "[Leiloeiro]";
	mes "Por "+$bid[@num]+"z "+$bidders$[@num]+" comprou isto de você";
	if($paid[@num]!=1) goto L_NoMoney;
	mes "Ele pagou certinho, então aqui está seu dinheiro.";
	next;
	set zeny,zeny+$bid[@num];
	set $paid[@num],2;
	goto L_End;
L_AlreadyPaid:
	mes "[Leiloeiro]";
	mes "Hmm, parece que você já foi pago por este lote, não está tentando me enganar está?";
	close;
L_Returned:
	mes "[Leiloeiro]";
	mes "Parece que nós já devolvemos seu item";
	close;
L_GiveBack:
	mes "[Leiloeiro]";
	mes "Parece que ninguém quis seu(s) item(s)";
	mes "Então você pode tê-los de volta agora";
	getitem $AItems[@num],$AAmount[@num];
	set $paid[@num],3;
	close;
L_NoMoney:
	mes "Sinto lhe dizer mas ainda não pagaram por seu(s) item(s)";
	mes "Você pode ir falar com ele agora, ou pode esperar.";
	mes "Se não vierem pagar até o próximo Leilão iniciar, você pode voltar para pegar seu(s) item(s) de volta";
	close;

// Check starts here for if you won any items during this auction, futher options follow

L_Won:
	set @num,0;
	mes "[Leiloeiro]";
L_WonLoop:
	if($bidders$[@num]==strcharinfo(0)) goto L_Winner;
L_CarryOn:
	set @num,@num+1;
	if(@num<10) goto L_WonLoop;
	mes "Sinto muito, você não ganhou nenhuma das ofertas, por favor tente na próxima vez";
	close;
L_Winner:
	if($paid[@num]>0) goto L_CarryOn;
	mes "Ora, parece que você ganhou um item";
	mes "Você ganhou "+$AAmount[@num]+" x "+getitemname($AItems[@num]);
	mes "E o seu lance foi "+$bid[@num]+"z para este lote";
	mes "O que você deseja fazer agora?";
	next;

// At this point they can choose from 3 options, to Buy the current lot
// see the next one they might have won, or give up that lot

	menu "Comprar este lote",-,"Ver próximo lote ganho",L_CarryOn,"Renunciar este lote",L_Back;
	if(zeny<$bid[@num]) goto L_2lowzeny;
	set zeny,zeny-$bid[@num];
	getitem $AItems[@num],$AAmount[@num];
	set $paid[@num],1;
	mes "[Leiloeiro]";
	mes "Nossa transação está completada.";
	mes "Vejo-o novamente em breve.";
	close;
L_2lowzeny:
	mes "[Leiloeiro]";
	mes "Desculpe, parece que você não tem zeny o suficiente, por favor adquira mais.";
	close;
L_Back:
	mes "[Leiloeiro]";
	mes "Você tem certeza que quer desistir deste lote?";
	next;
	menu "Sim",-,"Não",L_End;
	set $bid[@num],0;
	set $bidders$[@num],"ninguém";
	mes "[Leiloeiro]";
	mes "Está feito, você talvez que falar para "+$ANames$[@num]+" que ele pode vir e pegar seu(s) item(s)";
	close;

	close;

	mes "[Ajudante Leiloeiro]";
	set @num,0;
	if($Auction==0 && $AItemsB[@num]!=0) goto L_ItemRecover;
	if(($Auction==1 || $Auction ==2) && $AItems[@num]!=0) goto L_LLoop;
	if($Auction==3 && $AItems[@num]!=0) goto L_Win;
	mes "Ok, atualmente nós não temos nada no leilão.";
	mes "Por favor retorne no horário correto.";
	close;
L_LLoop:
	mes "Ok, atualmente nós temos :-";
L_Loop:
	mes $ANames$[@num]+" com "+$AAmount[@num]+" x "+getitemname($AItems[@num])+"('s)";
	set @num,@num+1;
	if($AItems[@num]==0) close;
	if(@num<10) goto L_Loop;
